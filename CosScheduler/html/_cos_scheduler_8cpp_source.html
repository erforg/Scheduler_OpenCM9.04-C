426 int8_t CosRunScheduler(void)
427 { // Scheduler version for Arduino and openCM:
428  Node_t *pt=NULL;
429  uint16_t t_Ticks;
430 
431  //DebugCode(_msg("RunScheduler,prio based\r\n"););
432 
433  pt = COS_GetTaskListRootPointer(); /* first task, highest prio */
434  while(1) /* loop forever */
435  { /* time to run? */
436  t_Ticks =millis();
437  /* time wrap around is ok, time difference will be right... */
438  if(((uint16_t)(t_Ticks - pt->task_pt->lastActivationTime_Ticks) >=
439  pt->task_pt->sleepTime_Ticks)&&
440  (pt->task_pt->state == TASK_STATE_READY))
441  { pt->task_pt->lastActivationTime_Ticks = t_Ticks;
442  /* when the task function runs to its very end, the task will be deleted:
443  it will be removed from the list, and the task struct will be freed,
444  i.e. pt->task_pt is no longer valid.
445  */
446  pt->task_pt->sleepTime_Ticks = 0; // Bugfix 22.10.2015: sleep time must be specified by task!
447  pt->task_pt->func(pt->task_pt); /* call task function, must not block! */
448  pt = COS_GetTaskListRootPointer(); /* next: check task with highest prio */
449  }
450  else
451  { pt = pt->next_pt; /* check next task in list */
452  if(NULL==pt)
453  { pt = COS_GetTaskListRootPointer(); /* treat linear list as ring list */
454  }
455  }
456  }
457  return 0;
458 }
