138 uint8_t COS_FifoCreate(CosFifo_t *q, uint8_t slotSize, uint8_t nSlots)
139 {
140  /* create buffer */
141  q->buffer = (char *) malloc(slotSize * nSlots * sizeof(char));
142  if(NULL == q->buffer)
143  { DebugCode(_msg("FifoCreate:malloc!"););
144  return -1;
145  }
146  q->maxSlots = nSlots;
147  q->slotSize = slotSize;
148  q->rIndex = 0; /* empty queue */
149  q->wIndex = 0;
150  q->usedSlots = 0;
151  if(0!= COS_SemCreate(&(q->rSema), 0)) // nothing to read yet
152  { DebugCode(_msg("FifoCreate:SemCreate!"););
153  return -1;
154  }
155  if(0!= COS_SemCreate(&(q->wSema), nSlots)) // all slots are still free
156  { DebugCode(_msg("FifoCreate:SemCreate!"););
157  return -1;
158  }
159  q->isInitialized = 1;
160  return 0;
161 }

327 int8_t _qWriteSingleSlot(CosFifo_t *q, const char *data)
328 { int8_t retval = 0;
329  if(q->isInitialized == 0)
330  { DebugCode(_msg("_qWriteSingleSlot:not init"););
331  return -1;
332  }
333  /* if FIFO is full, return 0 without writing any data */
334  if(q->usedSlots >= q->maxSlots)
335  { retval = 0; /* FIFO is full, don't write */
336  }
337  else /* at least one slot is free, write data */
338  { retval = 1;
339  memcpy(&(q->buffer[q->wIndex]), data, q->slotSize); /* copy to FIFO */
340  q->wIndex += q->slotSize; /* next slot */
341  q->wIndex %= (q->maxSlots * q->slotSize); /* circular buffer */
342  q->usedSlots += 1;
343  COS_SEM_SIGNAL(&(q->rSema)); // unblock tasks that wait for reading,
344  }
345  return retval;
346 }

380 int8_t _qReadSingleSlot(CosFifo_t *q, char *data)
381 { int8_t retval =0;
382  if(q->isInitialized == 0)
383  { DebugCode(_msg("_qReadSingleSlot:not init"););
384  return -1;
385  }
386  /* if FIFO is empty, nothing is read */
387  if (q->usedSlots <=0)
388  { retval = 0;
389  }
390  else /* at least one slot has data */
391  { retval = 1;
392  memcpy(data, &(q->buffer[q->rIndex]), q->slotSize); /* read from queue */
393  q->rIndex += q->slotSize; /* next slot to read */
394  q->rIndex %= (q->maxSlots * q->slotSize); /* circular buffer */
395  q->usedSlots -= 1;
396  COS_SEM_SIGNAL(&(q->wSema)); // unblock tasks that wait for writing
397  }
398  return retval;
399 }
