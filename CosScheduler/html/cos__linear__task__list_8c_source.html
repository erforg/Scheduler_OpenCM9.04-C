191  Node_t *_unlinkTaskFromTaskList(Node_t *root_pt, CosTask_t *task_pt)
192 { Node_t *pt=NULL;
193  Node_t *predecessor_pt=NULL;
194 
195  pt = _searchTaskInList(root_pt, task_pt);
196  if(NULL == pt)
197  { DebugCode(_msg("_unlink...task not found"););
198  return root_pt; /* don't change the list! */
199  }
200  /* is it the first node in the list? root_pt has to be changed... */
201  if(pt == root_pt)
202  { pt = pt->next_pt; /* second node in list */
203  free(root_pt); /* free the first node, don't touch the task! */
204  return pt; /* the old second element now is the first */
205  }
206  /* node exists, is not the first list element, should have a predecessor... */
207  predecessor_pt = _searchPredecessorTaskInList(root_pt, task_pt);
208  if(NULL == predecessor_pt)
209  { DebugCode(_msg("_unlink...pred.task not found"););
210  return root_pt; /* don't change the list! */
211  }
212  /* task node and its predecessor have been found. Unlink task node: */
213  predecessor_pt->next_pt = pt->next_pt;
214  free(pt); /* free the node, don't touch the task! */
215  return root_pt; /* old first list element still is the first */
216 }

442 void _sortLinearListPrio(Node_t *root_pt)
443 {
444 
445  int noOfSwaps;
446  CosTask_t *temp_pt;
447  Node_t *pt;
448 
449  if (root_pt==NULL) return; /* empty list */
450 
451  /* bubblesort, slow but simple... */
452  do
453  { noOfSwaps=0;
454  pt = root_pt;
455  while(pt->next_pt != NULL) // run through linear list
456  { if((pt->task_pt)->prio < ((pt->next_pt)->task_pt)->prio)
457  { /* swap the 'payloads' of the nodes*/
458  noOfSwaps++;
459  temp_pt = pt->task_pt; /* swap payload of nodes */
460  pt->task_pt = (pt->next_pt)->task_pt;
461  (pt->next_pt)->task_pt = temp_pt;
462  }
463  pt = pt->next_pt;
464  }
465  } while(noOfSwaps >0);
466 }
