276 static void _idleTask(CosTask_t *pt)
277 {
278  COS_TASK_BEGIN(pt);
279  while(1)
280  { if(cpuLoadCounter_g > 0)
281  { cpuLoadCounter_g--;
282  }
283  //DebugCode(_toggle_PD(5););
284  COS_TASK_SLEEP(pt,IDLE_TASK_PERIOD_TICKS);
285  }
286  COS_TASK_END(pt);
287 }

355 int8_t COS_InitTaskList(void)
356 {
357  //DebugCode(_msg("InitTaskList\r\n"););
358  root_g = NULL; /* empty task list */
359  /* task functions are kept in a linear list, that always has at least

367

369  return 0;

628 int8_t COS_RunScheduler(void)
629 {
630 
631 #if COS_PLATFORM == PLATFORM_RENESAS_RX63N
632 
633  Node_t *pt=NULL;
634  uint16_t t_Ticks;
635 
636  //DebugCode(_msg("RunScheduler,prio based\r\n"););
637 
638  pt = root_g; /* first task, highest prio */
639  while(1) /* loop forever */
640  { /* time to run? */
641  t_Ticks = _gettime_Ticks();
642  /* time wrap around is ok, time difference will be right... */
643  if(((uint16_t)(t_Ticks - pt->task_pt->lastActivationTime_Ticks) >=
644  pt->task_pt->sleepTime_Ticks)&&
645  (pt->task_pt->state == TASK_STATE_READY))
646  { pt->task_pt->lastActivationTime_Ticks = t_Ticks;
647  /* when the task function runs to its very end, the task will be deleted:
648  it will be removed from the list, and the task struct will be freed,
649  i.e. pt->task_pt is no longer valid.
650  */
651  pt->task_pt->sleepTime_Ticks = 0; // Bugfix 22.10.2015: must be specified by task!
652  pt->task_pt->func(pt->task_pt); /* call task function, must not block! */
653  pt = root_g; /* next: check task with highest prio */
654  }
655  else
656  { pt = pt->next_pt; /* check next task in list */
657  if(NULL==pt)
658  { pt = root_g; /* treat linear list as ring list */
659  }
660  }
661  }
662 #endif // COS_PLATFORM
663 
664  // not implemented on Arduino and openCM, use CosRunScheduler() instead
665 
666  return 0;
667 
668 }
669 #else
