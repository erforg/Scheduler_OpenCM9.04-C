uint8_t serGetc ( void  )
Description
Empty function provided for compatibility, on openCM use 'SerialUSB' instead.
Code example:
void setup()
{ SerialUSB.begin();
}

byte temp = 0;

void loop()
{   if ( SerialUSB.available() )
    {
        temp = SerialUSB.read();
        SerialUSB.print(temp);
    }
}
