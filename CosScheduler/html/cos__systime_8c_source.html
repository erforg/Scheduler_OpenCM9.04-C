186  void _initSystemTime(void)
187 {
188  // take from FreeRtos
189  /* switch off register protection
190  PRCR = protect register. */
191  SYSTEM.PRCR.WORD = 0xA50B;
192 
193  /* Modul-Stop-State of Compare-Match-Timer 0 is deactivated.
194  MSTP = module stop state, it is a Makro defined in iodefine.h. */
195  MSTP( CMT0 ) = 0;
196 
197  /* switch on register protection */
198  SYSTEM.PRCR.WORD = 0xA500;
199 
200  /* activate interrupt
201  CMT0 = compare match timer 0,
202  CMCR = compare match timer control register,
203  CMIE = compare match interrupt enable. */
204  CMT0.CMCR.BIT.CMIE = 1;
205 
206  /* set compare value for interrupt
207  CMCOR = compare match timer constant register. */
208  //CMT0.CMCOR = ( unsigned short ) ( ( ( configPCLKB_HZ / configTICK_RATE_HZ ) - 1 ) / 8 );
209  //CMT0.CMCOR = 0x5555; // 3.5 ms
210  //CMT0.CMCOR = 0x2222; // 1.5 ms
211  //CMT0.CMCOR = 0x1560; // 0.917 ms
212  //CMT0.CMCOR = 0x1600; // 0.956 ms
213  CMT0.CMCOR = 0x1700; // 0.99 ms
214 
215  /* PCLKB/8 is clock source
216  CKS = clock select. */
217  CMT0.CMCR.BIT.CKS = 0;
218 
219  /* set interrupt priority */
220  //IPR( CMT0, CMI0 ) = configKERNEL_INTERRUPT_PRIORITY;
221  IPR(CMT0,CMI0) = 2;
222 
223  /* activate interrupt in the ICU */
224  IEN( CMT0, CMI0 ) = 1;
225 
226  /* start the timer
227  CMSTR0 = compare match timer start register 0,
228  STR0 = count start 0. */
229  CMT.CMSTR0.BIT.STR0 = 1;
230  #if DEBUG
231  _LedInitPortDirections();
232  #endif
233 }
